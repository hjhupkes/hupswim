Private Function initReport() As Boolean
On Error GoTo fout
initReport = False

blUseFourTst = False


'Eerst kijken of tussentijden wel nodig...
Dim sql As String
sql = "SELECT nz([TY_Tijd4])+nz([ty_tijd3]) AS Expr1 FROM [~vUitslagMain] WHERE (((nz([TY_Tijd4])+nz([ty_tijd3]))>1));"

Dim rs As Recordset
Set rs = CurrentDb().OpenRecordset(sql)
rs.MoveFirst
If Not rs.EOF Then
  blUseFourTst = True
End If

Dim firstWedNr As Long, lastWedNr As Long
firstWedNr = Nz(DMin("TY_Wedstrijdnummer", "[~vUitslagMain]"))
lastWedNr = Nz(DMax("TY_Wedstrijdnummer", "[~vUitslagMain]"))

blEntireMeet = False
If (firstWedNr <> lastWedNr) Then
  blEntireMeet = True
End If

If (firstWedNr <= 0) Then
  Err.Raise 1, , "Geen tijden gevonden"
End If

Dim isFnd As Boolean
If Not wedLookup(firstWedNr, isFnd, firstWedInfo) Then
  Exit Function
End If
If Not isFnd Then
  Err.Raise 1, , "Kon wedstrijd: " & firstWedNr & " niet vinden"
End If


mCatCurWedNr = -1

Set rsTST = CurrentDb().OpenRecordset("fdtgTussenAfstanden")
rsTST.Index = "PrimaryKey"



initReport = True
Exit Function

fout:
If (Err = 3021) Then Resume Next
MsgBox Err & " initReport" & vbCrLf & Error$, 16
Exit Function
End Function

Private Sub scProgVarCatHeader_Format(Cancel As Integer, FormatCount As Integer)
On Error Resume Next

If dbUitslagRecsVerbergen() Then
  Cancel = True
  Exit Sub
End If


Dim curWed As Long
curWed = Nz([TY_Wedstrijdnummer])

If (curWed <> mCatCurWedNr) Then
  Dim dummy(0 To 0) As String
  If (Not catInfoGetListForWed(curWed, nMCatProgs, mCatProgNames, dummy, mCatProgOffs, mCatProgCatCnt, catInfo, False)) Then
    nMCatProgs = 0
    Exit Sub
  End If
  mCatCurWedNr = curWed
End If




Dim captSet As Boolean
captSet = False

If (nMCatProgs > 0) Then

  Dim prognr As String
  prognr = Nz(Me.[TY_Programmanummer])
  Dim progIdx As Integer
  progIdx = catInfoGetProgIdx(prognr, nMCatProgs, mCatProgNames)
  If (progIdx >= 0) Then
    Me.extBesProg.value = fmtCatLijst(catInfo, mCatProgOffs(progIdx), mCatProgCatCnt(progIdx))
    captSet = True
    
    Dim i As Integer
    Dim lastRecIdx As Integer
    lastRecIdx = 0
    For i = 0 To mCatProgCatCnt(progIdx) - 1
      If (Len(Trim(catInfo(mCatProgOffs(progIdx) + i).rec.recType)) > 0) Then
        With catInfo(mCatProgOffs(progIdx) + i).rec
          lastRecIdx = lastRecIdx + 1
          Me.Report.Controls("recType" & lastRecIdx).value = Trim(.recType)
          Me.Report.Controls("recNaam" & lastRecIdx).value = Trim(.naam)
          Me.Report.Controls("recTijd" & lastRecIdx).value = fmtSwimTime(.Tijd)
          Me.Report.Controls("recDatum" & lastRecIdx).value = fmtDateShort(.Datum)
          Me.Report.Controls("recPlaats" & lastRecIdx).value = Trim(.Plaats)
          Me.Report.Controls("recCat" & lastRecIdx).value = Trim(catInfo(mCatProgOffs(progIdx) + i).cat)
        End With
      End If
    Next i
    For i = lastRecIdx + 1 To MAX_N_VAR_CATS_PER_PROG
          Me.Report.Controls("recType" & i).value = ""
          Me.Report.Controls("recNaam" & i).value = ""
          Me.Report.Controls("recTijd" & i).value = ""
          Me.Report.Controls("recDatum" & i).value = ""
          Me.Report.Controls("recPlaats" & i).value = ""
          Me.Report.Controls("recCat" & i).value = ""
    Next i
  End If
End If
If Not captSet Then
  Cancel = True
End If
End Sub
