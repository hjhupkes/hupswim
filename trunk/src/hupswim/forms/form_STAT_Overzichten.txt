Private Sub handleView(ByVal rptOpenMode As Integer, ByVal rptNCopies As Integer, ByVal rxpFormat As Integer)
On Error GoTo fout


Dim dPrimStartDate As Date, dPrimEndDate As Date
Dim sPrimSeason As String
Dim optUseSecond As Boolean
Dim dSecStartDate As Date, dSecEndDate As Date
Dim sSecSeason As String
Dim optBoth25And50m As Boolean

dPrimStartDate = Nz(Me.fldPrimStartDate, #1/1/1900#)
dPrimEndDate = Nz(Me.fldPrimEndDate, #1/1/1900#)
sPrimSeason = Nz(Me.fldPrimSeason)
optUseSecond = Nz(Me.blUseSecondary)
dSecStartDate = Nz(Me.fldSecStartDate, #1/1/1900#)
dSecEndDate = Nz(Me.fldSecEndDate, #1/1/1900#)
sSecSeason = Nz(Me.fldSecSeason)
optBoth25And50m = Nz(Me.blBoth25And50)

Dim dAllStartdate As Date, dAllEnddate As Date
Dim iMaxCnt As Integer
dAllStartdate = Nz(Me.fldAllStartDate, #1/1/1900#)
dAllEnddate = Nz(Me.fldAllEndDate, #1/1/1900#)
iMaxCnt = Nz(Me.fldMaxCnt)



Dim optExcludeTst As Boolean, optExcludeDis As Boolean

optExcludeTst = Not Nz(Me.blIncludeTst)
optExcludeDis = Not Nz(Me.blIncludeDis)


Dim optRestrictLS As Boolean
Dim optOnlyActive As Boolean
Dim optRestrictAfSl As Boolean
Dim optRestrictWed As Boolean
Dim sStartnummer As String

optRestrictLS = Nz(Me.blRestrictLS)
optOnlyActive = Nz(Me.blOnlyActive)
sStartnummer = Nz(Me.fldStartnummer)
If (Not optRestrictLS) Then
  optOnlyActive = False
  sStartnummer = ""
End If

optRestrictAfSl = Nz(Me.blRestrictAfSl)
optRestrictWed = Nz(Me.blRestrictWed)


Dim optShowLevel As Boolean, optSortAchternaam As Boolean
optShowLevel = Nz(Me.blShowLevel)
optSortAchternaam = Nz(Me.blSortAchternaam)

Dim optCmpIsExtended As Boolean
optCmpIsExtended = Nz(Me.blCmpExtended)


Dim chc As Integer
chc = getChoice

Dim rptName As String



Select Case chc
  Case OPT_SN
    If Not timBuildList(True, dPrimStartDate, dPrimEndDate, sPrimSeason, _
      optUseSecond, dSecStartDate, dSecEndDate, sSecSeason, optRestrictAfSl, nAfSl, ordAf, ordSl, _
      optRestrictLS, lsCode, optOnlyActive, sStartnummer, optRestrictWed, nWeds, ordWed, _
      optExcludeDis, optExcludeTst) Then
        Exit Sub
    End If
    
    'Now prepare the report
    
      rptName = REP_SN
      
   If Not rxpSetTimeListOptions(optShowLevel, iMaxCnt, optSortAchternaam, optBoth25And50m, False) Then
     Exit Sub
   End If

    
    
    
  Case OPT_GT
    If Not timBuildList(False, dAllStartdate, dAllEnddate, "", _
      False, #1/1/1900#, #1/1/1900#, "", optRestrictAfSl, nAfSl, ordAf, ordSl, _
      optRestrictLS, lsCode, optOnlyActive, sStartnummer, optRestrictWed, nWeds, ordWed, _
      optExcludeDis, optExcludeTst) Then
        Exit Sub
    End If
    
    rptName = REP_GT
    
    If Not rxpSetTimeListOptions(optShowLevel, iMaxCnt, optSortAchternaam, optBoth25And50m, False) Then
      Exit Sub
    End If

    
  Case OPT_CMP
    If Not timBuildSeasonFastTimes(optCmpIsExtended) Then
      Exit Sub
    End If
  
    rptName = REP_SN_CMP
    
    If Not rxpSetCmpFastTimesOptions() Then
      Exit Sub
    End If

End Select

Dim vDummy(0 To 0) As Variant

If Not rptHandle(False, rptName, rptOpenMode, rptNCopies, rxpFormat, 0, vDummy) Then
  Exit Sub
End If


Exit Sub
fout:
MsgBox Err & " handleView" & vbCrLf & Error$, 16
Exit Sub

'debug
On Error GoTo 0
Resume Next
End Sub
