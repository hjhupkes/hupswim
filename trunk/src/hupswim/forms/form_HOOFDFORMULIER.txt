Private Sub checkNewVersion()
On Error Resume Next

Dim blNewVersionAvailable As Boolean, blNeedsManualCheck As Boolean, sErrMsg As String, sNewVersion As String
If Not appCheckNewVersion(blNewVersionAvailable, blNeedsManualCheck, sNewVersion, False, True, sErrMsg) Then
  Exit Sub
End If

If (blNewVersionAvailable) Then
  Me.btnLeden.SetFocus
  Me.btnScanNewVersion.Visible = False
  Me.btnNewVersion.Visible = True
  Me.btnNewVersion.Caption = "Upgraden naar versie " & sNewVersion
ElseIf (blNeedsManualCheck) Then
  Me.btnLeden.SetFocus
  Me.btnScanNewVersion.Visible = True
  Me.btnNewVersion.Visible = False
Else
  Me.btnLeden.SetFocus
  Me.btnScanNewVersion.Visible = False
  Me.btnNewVersion.Visible = False
End If



End Sub
