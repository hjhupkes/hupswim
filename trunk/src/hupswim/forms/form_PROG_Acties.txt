Private Sub btnCatBes_Click()
On Error GoTo fout

If MsgBox("Deze optie zal de huidige categorie beschrijvingen vervangen door de standaard categorie beschrijvingen die Hupprog automatisch aanmaakt." & Chr(13) & _
"Gebruik deze optie als u een lenex bestand hebt geimporteerd en de beschrijvingen voldoen niet." & Chr(13) & _
"Wilt u doorgaan?", vbYesNo + vbQuestion) = vbNo Then
  Exit Sub
End If
Dim rs As Recordset
Set rs = CurrentDb().OpenRecordset("select * from [dtProgrammanummers] where wedstrijdnr=" & waarde)
rs.MoveFirst
Do Until rs.EOF
  rs.Edit
  rs("Beschrijving categorie") = catBeschrijving(Nz(rs("Categorie")))
  rs.Update
  rs.MoveNext
Loop

appCleanRS rs

Exit Sub
fout:
If (Err = 3021) Then Resume Next ' no current record
MsgBox Err & " btnCatBes_click" & Error$, 16
End Sub
