Private Sub handleView(ByVal blDetails As Boolean)
On Error GoTo fout

If Not blBerekend Then
  Dim optRestrictAfSl As Boolean
  Dim optRestrictLS As Boolean
  Dim optUseLft As Boolean
  Dim optApart50m As Boolean
  Dim optPntMode As Integer
  
  Dim minDate As Date, startDate As Date, endDate As Date
  
  minDate = Nz(Me.datum0, #1/1/1900#)
  startDate = Nz(Me.datum1, #1/1/1900#)
  endDate = Nz(Me.datum2, #1/1/1900#)
  
  optRestrictAfSl = Nz(Me.blRestrictAfSl)
  optRestrictLS = Nz(Me.blRestrictLS)
  optApart50m = False
  optUseLft = Nz(Me.blUseLft)
  optPntMode = Nz(Me.pntMode)
  
  
  If Not lenDateIsValid(startDate) Then
    MsgBox "Voer een startdatum in", vbCritical
    Me.datum1.SetFocus
    Exit Sub
  End If
  
  If Not lenDateIsValid(endDate) Then
    MsgBox "Voer een einddatum in", vbCritical
    Me.datum2.SetFocus
    Exit Sub
  End If
    

  If Not timCalcVooruitgang(minDate, startDate, endDate, optApart50m, optPntMode, _
    optUseLft, nCat, ordCat, optRestrictLS, lsCode, _
    optRestrictAfSl, nAfSl, ordAf, ordSl) Then
    Exit Sub
  End If
  
  blBerekend = True
  
End If


If Not rxpSetVooruitgangOptions(Me.fldTitle) Then
  Exit Sub
End If

Dim vDummy(0 To 0) As Variant

If (blDetails) Then
  If Not rptHandle(False, REP_VrtDet, RPT_MODE_PREVIEW, -1, -1, 0, vDummy) Then
    Exit Sub
  End If
Else
  If Not rptHandle(False, REP_Vrt, RPT_MODE_PREVIEW, -1, -1, 0, vDummy) Then
    Exit Sub
  End If
End If


Exit Sub
fout:
MsgBox Err & " handleView" & vbCrLf & Error$, 16
Exit Sub
End Sub

Private Function saveVRTOpts(ByVal sName As String) As Boolean
On Error GoTo fout
saveVRTOpts = False


Dim vrtOpts As RXC_STAT_VRT_OPTS



vrtOpts.dRefdate = Nz(Me.datum0, #1/1/1900#)
vrtOpts.dStartDate = Nz(Me.datum1, #1/1/1900#)
vrtOpts.dEndDate = Nz(Me.datum2, #1/1/1900#)

vrtOpts.iVrtMode = Nz(Me.pntMode)

vrtOpts.blRestrictLS = Nz(Me.blRestrictLS)
vrtOpts.lsCode = lsCode

vrtOpts.blUseLft = Nz(Me.blUseLft)
vrtOpts.blRestrictAfSl = Nz(Me.blRestrictAfSl)

vrtOpts.sTitle = Nz(Me.fldTitle)




If Not rxcStoreStatVRTOpts(sName, vrtOpts, nAfSl, ordAf, ordSl, nCat, ordCat) Then
  Exit Function
End If


saveVRTOpts = True
Exit Function
fout:
MsgBox Err & " saveVRTOpts" & vbCrLf & Error$, 16
Exit Function
End Function




Private Function loadVRTOpts(ByVal sName As String) As Boolean
On Error GoTo fout
loadVRTOpts = False


Dim vrtOpts As RXC_STAT_VRT_OPTS

If Not rxcLoadStatVRTOpts(sName, vrtOpts, nAfSl, ordAf, ordSl, nCat, ordCat) Then
  Exit Function
End If

Me.datum0 = IIf(lenDateIsValid(vrtOpts.dRefdate), vrtOpts.dRefdate, Null)
Me.datum1 = IIf(lenDateIsValid(vrtOpts.dStartDate), vrtOpts.dStartDate, Null)
Me.datum2 = IIf(lenDateIsValid(vrtOpts.dEndDate), vrtOpts.dEndDate, Null)

Me.pntMode = vrtOpts.iVrtMode

Me.blRestrictLS = vrtOpts.blRestrictLS
lsCode = vrtOpts.lsCode

Me.blUseLft = vrtOpts.blUseLft

Me.blRestrictAfSl = vrtOpts.blRestrictAfSl

Me.fldTitle = vrtOpts.sTitle

loadVRTOpts = True
Exit Function
fout:
MsgBox Err & " loadVRTOpts" & vbCrLf & Error$, 16
Exit Function
End Function
