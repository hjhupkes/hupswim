Private Function getRptName() As String
On Error Resume Next
Dim chc As Integer

chc = getChoice

Select Case chc
  Case OPT_AANS
    If (Nz(Me.blAanMeet)) Then
      If (Nz(Me.blAanPerDeeln)) Then
        getRptName = REP_AAN_MEET_PER_DEELN
      Else
        getRptName = REP_AAN_MEET
      End If
    ElseIf (Nz(Me.blAanKlein)) Then
      getRptName = REP_AAN_SMALL
    ElseIf (Me.blAanPerDeeln) Then
      getRptName = REP_AAN_PER_DEELN
    Else
      getRptName = REP_AAN
    End If
  Case OPT_AANS_ZWM
    getRptName = REP_AAN
  Case OPT_INS
    If (Nz(Me.blInsKlein)) Then
      getRptName = REP_INS_SMALL
    Else
      getRptName = REP_INS
    End If
  Case OPT_INS_MEET
    getRptName = REP_INS_MEET
  Case OPT_UIT
    getRptName = REP_UIT
  Case OPT_UIT_PP
    getRptName = REP_UIT_PP
  Case OPT_UIT_TD
    getRptName = REP_UIT_TD
  Case OPT_UIT_MULTI
    getRptName = REP_UIT_MULTI
  Case OPT_PNT_DET, OPT_PNT_FIN
    If (Nz(Me.blPntKlein)) Then
      getRptName = REP_PNT_VER_KLEIN
    Else
      getRptName = REP_PNT_VER
    End If
  Case OPT_PNT_TEL
    getRptName = REP_TEL
  Case OPT_FAC_OV
    getRptName = REP_FAC_OVZ
  Case OPT_FAC_IND
    Dim optFacSmall As Boolean
    optFacSmall = Nz(Me.blFacKlein)
    If (optFacSmall) Then
      getRptName = REP_FAC_IND_SMALL
    Else
      getRptName = REP_FAC_IND
    End If
  Case OPT_AANS_MULTI
    getRptName = REP_AAN
  Case OPT_VBLD
    getRptName = REP_VLB_AAN
  Case OPT_WED_OV
    getRptName = REP_WED_OV
End Select
  


End Function



Private Function prepareRpt(ByVal chc As Integer) As Boolean
On Error GoTo fout
prepareRpt = False


Select Case chc
  Case OPT_AANS
    Dim optAansMeet As Boolean
    optAansMeet = Nz(Me.blAanMeet)
    If Not opBuildOpstelling(False, #1/1/1900#, #1/1/1900#, waarde(), optAansMeet, False) Then
      Exit Function
    End If
    Dim optStreep As Boolean, optSmallFont As Boolean, optHideStart As Boolean
    optStreep = Nz(Me.blAanStreep)
    optSmallFont = Nz(Me.blAanSmallFont)
    optHideStart = Nz(Me.blAanHideStartnummer)
    If Not rxpSetAanschrijvingOptions(optStreep, False, optSmallFont, optHideStart, False) Then
      Exit Function
    End If
  Case OPT_AANS_ZWM
    Dim optNoFootText As Boolean
    optNoFootText = Nz(Me.blNoVoetTekst)
    optSmallFont = Nz(Me.blAanZwmSmallFont)
    
    If Not opBuildOpstelling(False, #1/1/1900#, #1/1/1900#, waarde(), False, False) Then
      Exit Function
    End If
    
    If Not rxpSetAanschrijvingOptions(False, True, optSmallFont, False, optNoFootText) Then
      Exit Function
    End If
      
  Case OPT_INS
  
    optSmallFont = Nz(Me.blInsSmallFont)
  
    If Not opBuildOpstelling(False, #1/1/1900#, #1/1/1900#, waarde(), False, False) Then
      Exit Function
    End If
    
    If Not rxpSetAanschrijvingOptions(False, False, optSmallFont, False, False) Then
      Exit Function
    End If
    
  Case OPT_INS_MEET
  
    optSmallFont = Nz(Me.blInsMeetSmallFont)
  
    If Not opBuildOpstelling(False, #1/1/1900#, #1/1/1900#, waarde(), True, False) Then
      Exit Function
    End If
    
    If Not rxpSetAanschrijvingOptions(False, False, optSmallFont, False, False) Then
      Exit Function
    End If
    
  Case OPT_UIT, OPT_UIT_PP
    Dim optUitEntireMeet As Boolean, optUitShowNG As Boolean, optUitShowNGZA As Boolean
    optUitEntireMeet = Nz(Me.blUitEntireMeet)
    optUitShowNG = Nz(Me.blUitShowNG)
    optUitShowNGZA = Nz(Me.blUitShowNGZA)
    
    If Not uitCreate(waarde(), wed_MeetMainNr(), optUitEntireMeet, optUitShowNG, optUitShowNGZA) Then
      Exit Function
    End If
    
  Case OPT_UIT_TD
    'need to do nothing
    
  Case OPT_UIT_MULTI
    'still need to setup
    Dim optUitStartDate As Date, optUitEndDate As Date
    optUitStartDate = Nz(Me.fldUitStartDate, #1/1/1900#)
    optUitEndDate = Nz(Me.fldUitEndDate, #1/1/2099#)
    
    Dim optUitMultiShowNG As Boolean, optUitMultiShowNGZA As Boolean
    
    optUitMultiShowNG = Nz(Me.blUitMultiShowNG)
    optUitMultiShowNGZA = Nz(Me.blUitMultiShowNGZA)
    
    If Not timBuildTimeBijzList(optUitStartDate, optUitEndDate, optUitMultiShowNG, optUitMultiShowNGZA) Then
      Exit Function
    End If
    
    If Not rxpSetUitMultiOptions() Then
      Exit Function
    End If
    
  Case OPT_PNT_DET
    Dim sFirstProg As String, sLastProg As String
    sFirstProg = Nz(Me.pntProgStart)
    sLastProg = Nz(Me.pntProgEind)
    
    If Not pntCalc(waarde(), -1, PNT_CALC_MODE_RESULT) Then
      Exit Function
    End If
    
    If Not rxpSetVerPntUitslagOptions("", sFirstProg, sLastProg, False, "", False) Then
      Exit Function
    End If
  
  Case OPT_PNT_FIN
    If Not pntCalc(waarde(), -1, PNT_CALC_MODE_RESULT) Then
      Exit Function
    End If
    
    If Not rxpSetVerPntUitslagOptions("", "", "", True, "", False) Then
      Exit Function
    End If
  
  Case OPT_PNT_TEL
    If Not pntCalc(waarde(), -1, PNT_CALC_MODE_RESULT) Then
      Exit Function
    End If
    
    If Not rxpSetVerPntUitslagOptions("", "", "", False, ver(), False) Then
      Exit Function
    End If
  
  Case OPT_FAC_OV
    Dim optFacEntireMeet As Boolean
    optFacEntireMeet = Nz(Me.blFacEntireMeet)
    
    Dim optShowFee As Boolean
    optShowFee = Nz(Me.blShowFee)
    
    If Not facCalcFees(waarde(), optFacEntireMeet) Then
      Exit Function
    End If
    
    If Not rxpSetFaktuurOptions(False, "", dbDeelnemerLijstStartsTonen, optShowFee) Then
      Exit Function
    End If
     
     
  Case OPT_FAC_IND
  
    optFacEntireMeet = Nz(Me.blFacEntireMeet)
    
    optShowFee = Nz(Me.blShowFee)
    
    If Not facCalcFees(waarde(), optFacEntireMeet) Then
      Exit Function
    End If
    
    If Not rxpSetFaktuurOptions(False, "", dbDeelnemerLijstStartsTonen, optShowFee) Then
      Exit Function
    End If
     
    
  Case OPT_AANS_MULTI
  
    Dim dOVStartDate As Date, dOVEndDate As Date
    
    dOVStartDate = Nz(Me.fldOvStartDate, #1/1/1900#)
    dOVEndDate = Nz(Me.fldOvEndDate, #1/1/1900#)
    
    If (Not lenDateIsValid(dOVStartDate)) Then
      Err.Raise 1, , "Vul een startdatum in"
    End If
    
    If (Not lenDateIsValid(dOVEndDate)) Then
      Err.Raise 1, , "Vul een einddatum in"
    End If
    
    optSmallFont = Nz(Me.blAanMultiSmallFont)
    
    If Not opBuildOpstelling(True, dOVStartDate, dOVEndDate, -1, False, False) Then
      Exit Function
    End If
    
    If Not rxpSetAanschrijvingOptions(False, True, optSmallFont, False, False) Then
      Exit Function
    End If
  Case OPT_VBLD, OPT_WED_OV
    dOVStartDate = Nz(Me.fldOvStartDate, #1/1/1900#)
    dOVEndDate = Nz(Me.fldOvEndDate, #1/1/1900#)
    
    If (Not lenDateIsValid(dOVStartDate)) Then
      Err.Raise 1, , "Vul een startdatum in"
    End If
    
    If (Not lenDateIsValid(dOVEndDate)) Then
      Err.Raise 1, , "Vul een einddatum in"
    End If
    
    'Only output stats
    If Not opBuildOpstelling(True, dOVStartDate, dOVEndDate, -1, False, True) Then
      Exit Function
    End If
    
    If (chc = OPT_WED_OV) Then
      optSmallFont = Nz(Me.blOvWedSmallFont)
      If Not rxpSetAanschrijvingOptions(False, False, optSmallFont, False, False) Then
        Exit Function
      End If
    End If
  
    
    
End Select



prepareRpt = True
Exit Function
fout:
MsgBox Err & " prepareRpt" & vbCrLf & Error$, 16
Exit Function
End Function


Private Sub handleView(ByVal rptOpenMode As Integer, ByVal rptNCopies As Integer, ByVal rxpFormat As Integer)
On Error GoTo fout



If Not prepareRpt(getChoice()) Then
  Exit Sub
End If

Dim rptName As String
rptName = getRptName()

Dim vDummy(0 To 0) As Variant

If Not rptHandle(False, rptName, rptOpenMode, rptNCopies, rxpFormat, 0, vDummy) Then
  Exit Sub
End If


Exit Sub
fout:
MsgBox Err & " handleView" & vbCrLf & Error$, 16
Exit Sub

'debug
On Error GoTo 0
Resume Next
End Sub
