Private Sub initView()
On Error GoTo fout

Dim lExpID As Long

lExpID = CLng(Nz(Me.openArgs))

Dim sql As String

sql = "SELECT hpoRptExpListCurPars.*, hpoRptExpListPars.* " & _
 "FROM hpoRptExpListCurPars INNER JOIN hpoRptExpListPars ON " & _
 "hpoRptExpListCurPars.SP_P_ID = hpoRptExpListPars.P_ID " & _
 "WHERE hpoRptExpListCurPars.SP_EXP_ID=" & lExpID & ";"


Me.RecordSource = sql


Exit Sub
fout:
MsgBox Err & " " & Error$, 16
End Sub

Private Sub Form_BeforeUpdate(Cancel As Integer)
On Error Resume Next
Dim tpParType As Long, tpReq As Boolean
tpParType = Nz([P_Type])
tpReq = Nz([p_requird])

Dim tpVal As String
tpVal = Nz([SP_Value])

Dim sInvMsg As String

If Not rxlIsParOK(tpVal, tpParType, tpReq, sInvMsg) Then
  MsgBox "Parameter ongeldig!" & vbCrLf & vbCrLf & sInvMsg, vbCritical
  Cancel = True
  Exit Sub
End If



End Sub
