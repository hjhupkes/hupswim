Private Sub btnWijzig_Click()
On Error Resume Next
Dim tpFName As String

If Not DirectoryExists(extractFilePath(tpFName)) Then
  tpFName = directory_goedmaak(dbdir()) & "\*.htm*"
End If

If Not fileNamePromptUser(FILE_DLG_SAVE, "Navigatie-html bestand ", "HTML bestanden (*.htm, *.html)|*.htm*", tpFName, True, True) Then
  Exit Sub
End If

If (tpFName <> "") Then
  If (Right(tpFName, 4) <> ".htm") And (Right(tpFName, 5) <> ".html") Then
    tpFName = tpFName & ".html"
  End If
  Me.EL_NAV_FileName = hzn(tpFName)
  saveChanges
End If
