Private Function slag2(x) As String
On Error Resume Next
Select Case x
    Case "OW"
     slag2 = "Onderwater"
    Case "RUG"
        slag2 = "Rugslag"
    Case "SCHOOL"
        slag2 = "Schoolslag"
    Case "VLINDER"
        slag2 = "Vlinderslag"
    Case "VRIJ"
        slag2 = "Vrijeslag"
    Case "WISSEL"
        slag2 = "Wisselslag"
End Select
End Function

Private Function fmtRnk(ByVal lRnk As Long) As String
On Error Resume Next
fmtRnk = ""
If (lRnk > 0) And (lRnk < 10000) Then
  If (lRnk = 1) Then
    fmtRnk = "1ste"
  Else
    fmtRnk = lRnk & "de"
  End If
End If
End Function
