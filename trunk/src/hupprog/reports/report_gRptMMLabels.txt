Private Function initReport() As Boolean
On Error GoTo fout
initReport = False

If Not initFormat() Then
  Exit Function
End If

Set rsDetails = CurrentDb().OpenRecordset("~mm_labels_details")
rsDetails.Index = "PrimaryKey"

initReport = True
Exit Function
fout:
MsgBox Err & " initReport" & vbCrLf & Error$, 16
Exit Function
End Function

Private Function initFormat() As Boolean
On Error GoTo fout

initFormat = False

Dim szWidthTot As Long
Dim szHeightTot As Long

Dim szPhysMarginW As Long
szPhysMarginW = 0.7 * 567
Dim szPhysMarginH As Long
szPhysMarginH = 0.7 * 567

Dim szIntMarginW As Long
szIntMarginW = szPhysMarginW
Dim szIntMarginH As Long
szIntMarginH = szPhysMarginH
'Constraint: 2*szActMargin > szPhysMargin if not blSkipTopBottom, otherwise unconstrained



'get A4 paper dimensions
szWidthTot = 567 * 21#
szHeightTot = 567 * 29.7

Me.Report.Width = szWidthTot - 2 * szPhysMarginW - 0.02 * 567
Me.scDetail.Height = szHeightTot - 2 * szPhysMarginH - 0.02 * 567


'now get some layout options.
Dim nVert As Long, nHrz As Long, blSkipTopBottom As Boolean, blSkipLeftRight As Boolean
nHrz = Nz(DLookup("MMO_HrzCnt", "gMailMergeOptions"))
nVert = Nz(DLookup("MMO_VertCnt", "gMailMergeOptions"))
blSkipTopBottom = Nz(DLookup("MMO_blSkipTopBottom", "gMailMergeOptions"))
blSkipLeftRight = Nz(DLookup("MMO_blSkipLeftRight", "gMailMergeOptions"))
nLabsPerPage = nHrz * nVert


Dim sFontName As String
Dim lFontSize As Integer

sFontName = "Arial"
lFontSize = 8

If (nHrz > 4) Then
  sFontName = "Arial Narrow"
End If



If (nVert > 8) Then
  szIntMarginH = 0.39 * 567
  
  If (blSkipTopBottom) Then
    szIntMarginH = 0.15 * 567
  End If
End If



If (nHrz > 3) Then
  szIntMarginW = 0.38 * 567
  
  If (blSkipLeftRight) Then
    szIntMarginW = 0.15 * 567
  End If
End If

'Get the dimensions.
Dim szWidthLab As Long, szHeightLab As Long
szWidthLab = szWidthTot / nHrz
szHeightLab = szHeightTot / nVert


'Get the text dimensions
Dim szWidthTxt As Long, szHeightTxt As Long
Dim szLeftTxt As Long, szTopTxt As Long

szTopTxt = szIntMarginH  'the standard margins
szLeftTxt = szIntMarginW 'the standard margins

szHeightTxt = szHeightLab - 2 * szIntMarginH - 0.05 * 567
If (szHeightTxt > 4 * 567 * 0.45) Then
  szHeightTxt = 4 * 567 * 0.45
  szTopTxt = 0.5 * (szHeightLab - szHeightTxt)
End If

szWidthTxt = szWidthLab - 2 * szIntMarginW - 0.05 * 567
If (szWidthTxt > 567 * 4.95) Then
  szWidthTxt = 567 * 4.95
  szLeftTxt = 0.5 * (szWidthLab - szWidthTxt)
End If


If (szHeightTxt < 4 * 0.41 * 567) Then
  lFontSize = 7
End If
If (szHeightTxt > 4 * 0.46 * 567) Then
  lFontSize = 9
End If



Dim ctl As Control
For Each ctl In Me.scDetail.Controls
  ctl.FontName = sFontName
  ctl.FontSize = lFontSize
Next ctl
  


'Now perform the layouting
Dim ix  As Integer, iy As Integer
For ix = 1 To nHrz
  For iy = 1 To nVert
    Dim ictl As Integer
    ictl = (iy - 1) * nHrz + ix
   
    Me.Controls("lb_line1_" & ictl).Width = 1
    Me.Controls("lb_line1_" & ictl).Height = 1
    Me.Controls("lb_line2_" & ictl).Width = 1
    Me.Controls("lb_line2_" & ictl).Height = 1
    Me.Controls("lb_line3_" & ictl).Width = 1
    Me.Controls("lb_line3_" & ictl).Height = 1
    Me.Controls("lb_line4_" & ictl).Width = 1
    Me.Controls("lb_line4_" & ictl).Height = 1
    
    
    Dim blShouldSkip As Boolean
    blShouldSkip = False
    
    If ((iy = 1) Or (iy = nVert)) And (blSkipTopBottom) Then
      blShouldSkip = True
    End If
    
    If ((ix = 1) Or (ix = nHrz)) And (blSkipLeftRight) Then
      blShouldSkip = True
    End If
    
    If blShouldSkip Then
    
      'skip this label.
      Me.Controls("lb_line1_" & ictl).Visible = False
      Me.Controls("lb_line2_" & ictl).Visible = False
      Me.Controls("lb_line3_" & ictl).Visible = False
      Me.Controls("lb_line4_" & ictl).Visible = False
      
    Else
    
      Dim lLeftIntended As Long
      lLeftIntended = (ix - 1) * szWidthLab + szLeftTxt - szPhysMarginW
      If (lLeftIntended < 0) Then
        lLeftIntended = 0
      End If
      Me.Controls("lb_line1_" & ictl).Left = lLeftIntended
      
      Dim lWidthIntended As Long
      lWidthIntended = szWidthTxt - 0.05 * 567
      If (Me.Controls("lb_line1_" & ictl).Left + lWidthIntended >= Me.Width) Then
        lWidthIntended = Me.Width - Me.Controls("lb_line1_" & ictl).Left - 0.02 * 567
      End If
      Me.Controls("lb_line1_" & ictl).Width = lWidthIntended
      
      Dim lTopIntended As Long
      lTopIntended = (iy - 1) * szHeightLab + szTopTxt - szPhysMarginH
      If (lTopIntended < 0) Then
        lTopIntended = 0
      End If
      Me.Controls("lb_line1_" & ictl).Top = lTopIntended
      Me.Controls("lb_line1_" & ictl).Height = szHeightTxt * 0.25 - 0.05 * 567
      
      Me.Controls("lb_line2_" & ictl).Left = Me.Controls("lb_line1_" & ictl).Left
      Me.Controls("lb_line2_" & ictl).Width = Me.Controls("lb_line1_" & ictl).Width
      Me.Controls("lb_line2_" & ictl).Top = Me.Controls("lb_line1_" & ictl).Top + Me.Controls("lb_line1_" & ictl).Height
      Me.Controls("lb_line2_" & ictl).Height = Me.Controls("lb_line1_" & ictl).Height
      
      Me.Controls("lb_line3_" & ictl).Left = Me.Controls("lb_line1_" & ictl).Left
      Me.Controls("lb_line3_" & ictl).Width = Me.Controls("lb_line1_" & ictl).Width
      Me.Controls("lb_line3_" & ictl).Top = Me.Controls("lb_line2_" & ictl).Top + Me.Controls("lb_line2_" & ictl).Height
      Me.Controls("lb_line3_" & ictl).Height = Me.Controls("lb_line1_" & ictl).Height
      
      Me.Controls("lb_line4_" & ictl).Left = Me.Controls("lb_line1_" & ictl).Left
      Me.Controls("lb_line4_" & ictl).Width = Me.Controls("lb_line1_" & ictl).Width
      Me.Controls("lb_line4_" & ictl).Top = Me.Controls("lb_line3_" & ictl).Top + Me.Controls("lb_line3_" & ictl).Height
      
      Dim lHeightIntended As Long
      lHeightIntended = Me.Controls("lb_line1_" & ictl).Height
      If (Me.Controls("lb_line4_" & ictl).Top + lHeightIntended >= Me.scDetail.Height) Then
        lHeightIntended = Me.scDetail.Height - Me.Controls("lb_line4_" & ictl).Top - 0.02 * 567
      End If
      Me.Controls("lb_line4_" & ictl).Height = lHeightIntended
    End If 'if blShouldSkip
  Next iy
Next ix

Dim i As Integer
For i = nVert * nHrz + 1 To N_LABEL_ON_PAGE
  Me.Controls("lb_line1_" & i).Visible = False
  Me.Controls("lb_line2_" & i).Visible = False
  Me.Controls("lb_line3_" & i).Visible = False
  Me.Controls("lb_line4_" & i).Visible = False
Next i




initFormat = True

Exit Function
fout:
MsgBox Err & " initFormat" & vbCrLf & Error$, 16
Exit Function

'debug
On Error GoTo 0
Resume
End Function

Private Sub Report_Open(Cancel As Integer)
On Error Resume Next
If Not initReport() Then
  Cancel = True
End If
End Sub

Private Sub scDetail_Format(Cancel As Integer, FormatCount As Integer)
On Error GoTo fout

Dim i As Integer
For i = 1 To nLabsPerPage
  Dim tpIdx As Long
  tpIdx = Nz(Me.Controls("LB_Idx_" & i).value, -1)
  
  If (tpIdx >= 0) Then
    rsDetails.Seek "=", tpIdx
    If (rsDetails.NoMatch) Then
      Err.Raise 1, , "Internal error"
    End If
    
    Me.Controls("lb_line1_" & i).value = Nz(rsDetails("LB_Line1"))
    Me.Controls("lb_line2_" & i).value = Nz(rsDetails("LB_Line2"))
    Me.Controls("lb_line3_" & i).value = Nz(rsDetails("LB_Line3"))
    Me.Controls("lb_line4_" & i).value = Nz(rsDetails("LB_Line4"))
  Else
    Me.Controls("lb_line1_" & i).value = ""
    Me.Controls("lb_line2_" & i).value = ""
    Me.Controls("lb_line3_" & i).value = ""
    Me.Controls("lb_line4_" & i).value = ""
  End If
  
Next i


Exit Sub
fout:
MsgBox Err & " scDetail_format" & vbCrLf & Error$, 16
Exit Sub
'debug
On Error GoTo 0
Resume
End Sub
