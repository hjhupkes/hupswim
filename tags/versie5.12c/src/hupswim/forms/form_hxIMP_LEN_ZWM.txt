Private Sub selectAll(sel As Boolean)
On Error Resume Next
Dim rs As Recordset
Set rs = Me.RecordsetClone


DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20

rs.MoveFirst
Do Until rs.EOF
  rs.Edit
  rs("IMP_Import") = sel
  rs.Update
  rs.MoveNext
Loop

Me.Requery

End Sub

Private Sub cmdSelKnown_Click()
On Error Resume Next
Dim rs As Recordset
Set rs = Me.RecordsetClone


DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20

rs.MoveFirst
Do Until rs.EOF
  rs.Edit
    If (Nz(rs("imp_isZwmKnown"))) Then
      rs("IMP_Import") = True
    Else
      rs("IMP_Import") = False
    End If
  rs.Update
  rs.MoveNext
Loop

Me.Requery

End Sub

On Error Resume Next

End Sub

Private Sub btnSelAll_Click()
On Error Resume Next
selectAll (True)
