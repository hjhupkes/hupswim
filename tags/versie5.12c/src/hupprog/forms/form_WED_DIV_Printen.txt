Private Function getChoice() As Integer
On Error Resume Next

  getChoice = Nz(optRepPrg.value, -1)
ElseIf (tabctl.value = tbPunten.PageIndex) Then
  getChoice = Nz(optRepPnt.value, -1)
ElseIf (tabctl.value = tbJury.PageIndex) Then
  getChoice = Nz(optRepJry.value, -1)
ElseIf (tabctl.value = tbStartkaarten.PageIndex) Then
  getChoice = Nz(optRepStk.value, -1)
ElseIf (tabctl.value = tbStartkaartenExt.PageIndex) Then
  getChoice = Nz(optRepStkExt.value, -1)
ElseIf (tabctl.value = tbFacturen.PageIndex) Then
  getChoice = Nz(optRepFac.value, -1)
ElseIf (tabctl.value = tbOverig.PageIndex) Then
  getChoice = Nz(optRepOvg.value, -1)
End If

End Function

Private Function getRptName() As String
On Error GoTo fout

getRptName = ""

Dim tpS As String

tpS = ""

Dim choice As Integer
choice = getChoice


Dim blFacKlein As Boolean
blFacKlein = Me.blFacKlein




Select Case choice
  Case OPT_PRG
    tpS = REP_PRG
  Case OPT_UTB
    tpS = REP_UTB
  Case OPT_UTL
    tpS = REP_UTL
  Case OPT_STL
    Dim blTwoCol As Boolean
    Dim blCat As Boolean
    blTwoCol = Nz(Me.blStlTwoCol, False)
    blCat = Nz(Me.blStlCat, False)
    If (blTwoCol And blCat) Then
      tpS = REP_STL_CAT_KLEIN
    ElseIf (blTwoCol) Then
      tpS = REP_STL_KLEIN
    ElseIf (blCat) Then
      tpS = REP_STL_CAT
    Else
      tpS = REP_STL
    End If
  Case OPT_PNT_DET, OPT_PNT_EIND
    If (pntType = PNT_TYPE_IND) Then
      tpS = REP_PNT_IND
    Else
      If (Nz(Me.blPntKlein)) Then
        tpS = REP_PNT_VER_KLEIN
      Else
       tpS = REP_PNT_VER
      End If
    End If
  Case OPT_TEL
    tpS = REP_TEL
  Case OPT_JRY
    tpS = REP_JRY
  Case OPT_DLV
    tpS = REP_DLV
  Case OPT_STK
    tpS = REP_STK
  Case OPT_KMP
    tpS = REP_KMP
  Case OPT_STK_EMPTY
    tpS = REP_STK
  Case OPT_STK_EXT_PERS
    tpS = REP_STK_EXT_PERS
  Case OPT_STK_EXT_EST_VOOR
    tpS = REP_STK_EXT_EST_VOOR
  Case OPT_STK_EXT_EST_ACHT
    tpS = REP_STK_EXT_EST_ACHT
  Case OPT_MDL
    tpS = REP_MM_LAB
  Case OPT_LENGTE_PROG
    tpS = REP_LNG_PROG
  Case OPT_SPD_OORK
    tpS = REP_SPD_OORK
  Case OPT_LNCAP
    tpS = REP_LNCAP
  Case OPT_FAC_OVZ
    tpS = REP_FAC_OVZ
  Case OPT_FAC_IND
    If (blFacKlein) Then
      tpS = REP_FAC_IND_KLEIN
    Else
      tpS = REP_FAC_IND
    End If
End Select


getRptName = tpS

Exit Function
fout:
MsgBox Err & " getRptName" & vbCrLf & Error$, 16
Exit Function
End Function



Private Function prepareRpt() As Boolean
On Error GoTo fout
prepareRpt = False

Dim choice As Integer
choice = getChoice

'First some checks

If (choice = OPT_PNT_DET) Or (choice = OPT_PNT_EIND) Or (choice = OPT_TEL) Then
  If (pntOrderID = -1) And Nz(Forms![wed_master]![WE_Vervangende_tijd]) = "N" Then
    Err.Raise 1, , "Puntentelling niet van toepassing."
  End If
End If

If (choice = OPT_TEL) Then
  If Nz(Me.ver) = "" Then
     Me.ver.SetFocus
     Err.Raise 1, , "U moet een vereniging kiezen.", 16
   End If
End If


Dim sStartProg As String, sEndProg As String
sStartProg = Nz(Me.prog1)
sEndProg = Nz(Me.prog2)




Select Case choice
  Case OPT_PRG, OPT_STL
    If wedIsProgRangeEmpty(waarde(), sStartProg, sEndProg) Then
      Err.Raise 1, , "Het opgegeven programmanummer bereik bevat geen programmanummers"
    End If
    
    If Not progCreate(False, waarde()) Then
      Exit Function
    End If
    
    If Not rxpSetPVOptions(waarde(), False, sStartProg, sEndProg, False, "", False, -1) Then
      Exit Function
    End If
  Case OPT_UTB, OPT_UTL
    If wedIsProgRangeEmpty(waarde(), sStartProg, sEndProg) Then
      Err.Raise 1, , "Het opgegeven programmanummer bereik bevat geen programmanummers"
    End If
    
    If Not uitslagCreate(waarde()) Then
      Exit Function
    End If
    
    Dim optRestrictRnk As Boolean, lMaxRnk As Long
    optRestrictRnk = False
    lMaxRnk = -1
    
    If (choice = OPT_UTL) Then
      optRestrictRnk = Nz(Me.blRestrictRnk)
      lMaxRnk = Nz(Me.fldMaxRnk)
    End If
    
    If Not rxpSetPVOptions(waarde(), False, sStartProg, sEndProg, False, "", optRestrictRnk, lMaxRnk) Then
      Exit Function
    End If
  Case OPT_PNT_DET, OPT_PNT_EIND, OPT_TEL
    If Not pntCalc(pntWedNr, pntOrderID, PNT_CALC_MODE_RESULT) Then
      Exit Function
    End If
    
    If (pntType <> PNT_TYPE_IND) Then
      'Apply some settings
      If choice = OPT_PNT_EIND Then
        If Not rxpSetVerPntUitslagOptions("", "", "", True, "", False) Then
          Exit Function
        End If
      ElseIf (choice = OPT_TEL) Then
        'apply single ver
        If Not rxpSetVerPntUitslagOptions("", "", "", False, Nz(Me.ver), False) Then
          Exit Function
        End If
      ElseIf (choice = OPT_PNT_DET) Then
        'apply prog range
        If Not rxpSetVerPntUitslagOptions("", Nz(Me.pntProgStart), Nz(Me.pntProgEind), False, "", Nz(Me.blPntHideEmptySubtotals, False)) Then
          Exit Function
        End If
      End If
    
    End If
    
    
  Case OPT_MDL
  
    If Not wedPrepareMedals(waarde(), Nz(Me.medEntireMeet), Nz(Me.medNHrz), Nz(Me.medNVert)) Then
      Exit Function
    End If
  
 
  Case OPT_FAC_IND, OPT_FAC_OVZ
    Dim optFacEntireMeet As Boolean
    optFacEntireMeet = Nz(Me.blFacEntireMeet)
    
    Dim optFacSingleVer As Boolean
    optFacSingleVer = Nz(Me.blFacSingleVer)
    
    Dim sFacSingleVer As String
    sFacSingleVer = Nz(Me.cmbFacVer)
    
    Dim optShowFee As Boolean
    optShowFee = Nz(Me.blShowFee)
    
    If (optFacSingleVer And (sFacSingleVer = "")) Then
      Err.Raise 1, , "Kies een vereniging."
    End If
    
    
    If (wed_IsMeet() And optFacEntireMeet) Then
      If Not facCalcFees(wed_MeetMainNr, True) Then
        Exit Function
      End If
    Else
      If Not facCalcFees(CLng(waarde()), False) Then
        Exit Function
      End If
    End If
    
    If Not rxpSetFaktuurOptions(optFacSingleVer, sFacSingleVer, dbDeelnemerLijstStartsTonen, optShowFee) Then
      Exit Function
    End If
    
  Case OPT_STK
  
    Dim optStkIncludePers As Boolean, optStkIncludeEst As Boolean
    Dim optStkIncludeStartzwemmers As Boolean, optStkIncludeReserve As Boolean
    Dim optStkEntireMeet As Boolean
    Dim optStkSortBaan As Boolean, optStkRestrictVer As Boolean, optStkRestrictProg As Boolean
    Dim optStkRestrictBn As Boolean
    
    Dim optStkVer As String, optStkFirstProg As String, optStkLastProg As String
    Dim optStkBaan As Integer
    
    optStkIncludePers = Nz(Me.blStkPers)
    optStkIncludeEst = Nz(Me.blStkEst)
    
    optStkIncludeStartzwemmers = Nz(Me.blStkStartzwemmer)
    optStkIncludeReserve = Nz(Me.blStkIncludeRes)
    
    optStkSortBaan = Nz(Me.blStkSortBaan)
    optStkRestrictVer = Nz(Me.blStkRestrictVer)
    
    optStkVer = Nz(Me.cmbStkVer)
    
    optStkRestrictBn = Nz(Me.blStkRestrictBaan)
    optStkBaan = Nz(Me.fldStkBaan)
    
    optStkEntireMeet = Nz(Me.blStkEntireMeet)
    optStkRestrictProg = Nz(Me.blStkRestrictProgs)
    optStkFirstProg = Nz(Me.fldStkFirstProg)
    optStkLastProg = Nz(Me.fldStkLastProg)
    
    
    
    If Not stkPrepare(waarde(), optStkEntireMeet, False, "", optStkRestrictProg, _
      optStkFirstProg, optStkLastProg, optStkRestrictVer, optStkVer, _
      optStkRestrictBn, optStkBaan, optStkIncludePers, optStkIncludeEst, optStkIncludeReserve, optStkIncludeStartzwemmers, _
      optStkSortBaan) Then
        Exit Function
    End If
       
      
      
  Case OPT_STK_EXT_PERS, OPT_STK_EXT_EST_VOOR, OPT_STK_EXT_EST_ACHT
  
    Dim optStkExtIncludeStartzwemmers As Boolean, optStkExtIncludeReserve As Boolean
    Dim optStkExtEntireMeet As Boolean
    Dim optStkExtSortBaan As Boolean, optStkExtRestrictVer As Boolean, optStkExtRestrictProg As Boolean
    Dim optStkExtRestrictBn As Boolean
    
    Dim optStkExtVer As String, optStkExtFirstProg As String, optStkExtLastProg As String
    Dim optStkExtBaan As Integer
    
    
    If (choice = OPT_STK_EXT_PERS) Then
      optStkExtIncludeStartzwemmers = Nz(Me.blStkExtStartzwemmer)
      optStkExtIncludeReserve = Nz(Me.blStkExtIncludeRes)
    Else
      optStkExtIncludeStartzwemmers = False
      optStkExtIncludeReserve = False
    End If
    
    optStkExtSortBaan = Nz(Me.blStkExtSortBaan)
    optStkExtRestrictVer = Nz(Me.blStkExtRestrictVer)
    
    optStkExtVer = Nz(Me.cmbStkExtVer)
    
    optStkExtRestrictBn = Nz(Me.blStkExtRestrictBaan)
    optStkExtBaan = Nz(Me.fldStkExtBaan)
    
    optStkExtEntireMeet = Nz(Me.blStkExtEntireMeet)
    optStkExtRestrictProg = Nz(Me.blStkExtRestrictProgs)
    optStkExtFirstProg = Nz(Me.fldStkExtFirstProg)
    optStkExtLastProg = Nz(Me.fldStkExtLastProg)
    
    
    
    If Not stkPrepare(waarde(), optStkExtEntireMeet, False, "", optStkExtRestrictProg, _
      optStkExtFirstProg, optStkExtLastProg, optStkExtRestrictVer, optStkExtVer, _
      optStkExtRestrictBn, optStkExtBaan, True, True, optStkExtIncludeReserve, optStkExtIncludeStartzwemmers, _
      optStkExtSortBaan) Then
        Exit Function
    End If
       
      
      
      

    
  
  Case OPT_KMP
  
    optStkEntireMeet = Nz(Me.blStkEntireMeet)
    optStkRestrictProg = Nz(Me.blStkRestrictProgs)
    optStkFirstProg = Nz(Me.fldStkFirstProg)
    optStkLastProg = Nz(Me.fldStkLastProg)
    
    If Not stkKmpPrepare(waarde(), optStkEntireMeet, False, "", optStkRestrictProg, optStkFirstProg, optStkLastProg) Then
      Exit Function
    End If
    
  Case OPT_STK_EMPTY
    If Not stkPrepareEmpty() Then
      Exit Function
    End If
    
  Case OPT_JRY
  
    Dim optIncludeVers As Boolean
    optIncludeVers = Not Nz(Me.deelnver, False)
    
    If Not vbldPrepare(dbJurySorterenOpAlfabet(), optIncludeVers, waarde()) Then
      Exit Function
    End If
    
  Case OPT_LENGTE_PROG
    If Not wedCalcLength(waarde()) Then
      Exit Function
    End If
  Case OPT_SPD_OORK
    'need to do nothing
    
    
End Select


prepareRpt = True

Exit Function
fout:
MsgBox Err & " prepareRpt" & vbCrLf & Error$, 16
Exit Function
End Function


Private Sub printen(rptOpenMode As Integer, rxpFormat As Integer)
On Error GoTo fout

Dim rptName As String
rptName = getRptName()

If (rptName <> "") Then
  If Not prepareRpt() Then
    Exit Sub
  End If
End If

Dim dummy(0 To 0)
Dim dummyHandled As Boolean

Dim nEx As Integer
nEx = val(Nz(Me.fldNEx))


'do not need to prepare
If Not rptHandle(False, rptName, rptOpenMode, nEx, rxpFormat, 0, dummy) Then
  Exit Sub
End If


Exit Sub
fout:
MsgBox Err & " printen" & vbCrLf & Error$, 16
Exit Sub
End Sub
