Private Sub btnWijzig_Click()
On Error GoTo fout
DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20



Dim tpCurName As String, tpWedNr As Long
tpCurName = Nz([EL_Name])
tpWedNr = Nz([EL_Wedstrijdnummer])

Dim sInputString As String
sInputString = Nz(InputBox("Nieuwe naam (voor " & tpCurName & ") :"))


Dim tpNewName As String
tpNewName = sInputString

If (tpNewName = "") Then
  Exit Sub
End If

Dim sDummy As String, lDummy As Long, blExists As Boolean
If Not rxlLookup(tpWedNr, tpNewName, blExists, sDummy, lDummy, lDummy) Then
  Exit Sub
End If

If (blExists) Then
  Err.Raise 1, , "De nieuwe naam: " & tpNewName & " bestaat al."
End If

[EL_Name] = tpNewName


DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20


Exit Sub

fout:
MsgBox Err & " btnWijzig_click" & vbCrLf & Error$, 16
Exit Sub

End Sub


Private Sub btnWijzigLenexFileName_Click()
On Error Resume Next
Dim tpFName As String
tpFName = Nz(Me.EL_LEN_FileName)

If Not DirectoryExists(extractFilePath(tpFName)) Then
  tpFName = directory_goedmaak(dbdir()) & "\*.l?f"
End If

If Not fileNamePromptUser(FILE_DLG_SAVE, "Lenex bestand ", "Lenex bestanden (*.lef, *.lxf)|*.l?f", tpFName, True, True) Then
  Exit Sub
End If

If (tpFName <> "") Then
  If (Right(tpFName, 4) <> ".lef") And (Right(tpFName, 4) <> ".lxf") Then
    tpFName = tpFName & ".lxf"
  End If
  Me.EL_LEN_FileName = hzn(tpFName)
  
  DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20
  
End If



End Sub
