Private Sub Detail1_Format(Cancel As Integer, FormatCount As Integer)
On Error Resume Next
'For some reason, it is necessary to repeat this for every format pass when exporting to rtf
If db_narrow() Then
  Dim ctl As Control
  For Each ctl In Me.Section(acDetail).Controls
    ctl.FontName = "Arial Narrow"
  Next ctl
End If

If (FormatCount = 1) Then
  If (blShowQlfData) Then
    Dim tpVer As String, tpWed As Long, tpProg As String
    tpVer = Nz(Me.PL_Vereniging)
    tpWed = Nz(Me.PL_Wedstrijdnummer)
    tpProg = Nz(Me.PL_Programmanummer)
    
    Dim i As Integer
    For i = 1 To 27
      Me.Controls("instd" & i) = ""
      Me.Controls("insdat" & i) = ""
      Me.Controls("insbn" & i) = ""
      Me.Controls("inspl" & i) = ""
      
      Dim tpStart As String
      tpStart = Nz(Me.Controls("startnr" & i))
      If (tpStart <> "") Then
        rsEst.Seek "=", tpVer, tpWed, tpProg, tpStart
        If (Not rsEst.NoMatch) Then
          Dim tpQlfBn As Integer, tpQlfDat As Date, tpQlfPl As String, tpQlfTd As Double
          tpQlfTd = Nz(rsEst("PL_QLF_Tijd"))
          tpQlfDat = Nz(rsEst("PL_QLF_Datum"), #1/1/1900#)
          tpQlfPl = Nz(rsEst("PL_QLF_Plaats"))
          tpQlfBn = Nz(rsEst("PL_QLF_Baan"), 0)
          
          If (swtIsValid(tpQlfTd)) Then
            Me.Controls("instd" & i) = fmtSwimTime(tpQlfTd)
            Me.Controls("insdat" & i) = fmtDateShort(tpQlfDat)
            Me.Controls("insbn" & i) = fmtBaanShort(tpQlfBn)
            Me.Controls("inspl" & i) = tpQlfPl
          End If
        End If 'if not rsEst.nomatch
      End If 'if tpStart <> ""
    Next i
    
  End If
End If 'if formatCount = 1

End Sub


Private Function apply_Programmaopmaak(ProgrammaOpmaak As TProgrammaOpmaak)
On Error GoTo fout


If (blShowQlfData) Then
  If (ProgrammaOpmaak.FEstafDeeln.Size > 8) Then
    ProgrammaOpmaak.FEstafDeeln.Size = 8
  End If
End If


Dim margin As Long
margin = Me.Detail1.Height - Me.hulp.Top - Me.hulp.Height


applyfont Me.naam, ProgrammaOpmaak.FEstafKop, 0.501 * 567, True

Dim i As Integer
For i = 1 To 27
  applyfont Me.Controls("volnaam" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
  applyfont Me.Controls("startnr" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
  applyfont Me.Controls("instd" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
  applyfont Me.Controls("insbn" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
  applyfont Me.Controls("insdat" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
  applyfont Me.Controls("inspl" & CStr(i)), ProgrammaOpmaak.FEstafDeeln, 0.481 * 567, True
Next i

Dim beginhoogte
beginhoogte = Me.naam.Top + Me.naam.Height



For i = 1 To 27
  Me.Controls("volnaam" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
  Me.Controls("startnr" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
  Me.Controls("instd" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
  Me.Controls("insbn" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
  Me.Controls("insdat" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
  Me.Controls("inspl" & CStr(i)).Top = beginhoogte + (i - 1) * Me.Controls("volnaam1").Height
Next i
   
Me.hulp.Height = 0
Me.hulp.Top = Me.Controls("volnaam27").Top + Me.Controls("volnaam27").Height
Me.hulp.Height = Me.Section(acDetail).Height - margin - Me.hulp.Top
   
   
If (blShowQlfData) Then
  For i = 1 To 27
    Me.Controls("startnr" & CStr(i)).Left = 0
    Me.Controls("startnr" & CStr(i)).Width = 1.35 * 567
    
    Me.Controls("volnaam" & CStr(i)).Left = Me.Controls("startnr1").Left + Me.Controls("startnr1").Width + 0.02 * 567
    Me.Controls("volnaam" & CStr(i)).Width = 3.2 * 567
    
    Me.Controls("instd" & CStr(i)).Visible = True
    Me.Controls("instd" & CStr(i)).Left = Me.Controls("volnaam1").Left + Me.Controls("volnaam1").Width + 0.02 * 567
    Me.Controls("instd" & CStr(i)).Width = 1.05 * 567
    
    Me.Controls("insbn" & CStr(i)).Visible = True
    Me.Controls("insbn" & CStr(i)).Left = Me.Controls("instd1").Left + Me.Controls("instd1").Width + 0.02 * 567
    Me.Controls("insbn" & CStr(i)).Width = 0.5 * 567
    
    Me.Controls("insdat" & CStr(i)).Visible = True
    Me.Controls("insdat" & CStr(i)).Left = Me.Controls("insbn1").Left + Me.Controls("insbn1").Width + 0.02 * 567
    Me.Controls("insdat" & CStr(i)).Width = 1.25 * 567
    
    Me.Controls("inspl" & CStr(i)).Visible = True
    Me.Controls("inspl" & CStr(i)).Left = Me.Controls("insdat1").Left + Me.Controls("insdat1").Width + 0.02 * 567
    Me.Controls("inspl" & CStr(i)).Width = 1.8 * 567
  Next i
  
    'Total width: 9.15cm
Else
  'Apply normal layout
  For i = 1 To 27
    Me.Controls("startnr" & CStr(i)).Left = 0
    Me.Controls("startnr" & CStr(i)).Width = 1.702 * 567
    Me.Controls("volnaam" & CStr(i)).Left = 1.705 * 567
    Me.Controls("volnaam" & CStr(i)).Width = 4.894 * 567
    
    Me.Controls("instd" & CStr(i)).Visible = False
    Me.Controls("insbn" & CStr(i)).Visible = False
    Me.Controls("insdat" & CStr(i)).Visible = False
    Me.Controls("inspl" & CStr(i)).Visible = False
  Next i
End If
   
   


Exit Function
fout:

MsgBox Err & " " & Error$
Exit Function

End Function

On Error Resume Next


Dim rxpPnt As RXP_Pointer

Dim iDummy As Integer

If Not expPV_getStats(iDummy, rxpPnt) Then
  Exit Sub

argPVFmtName = rxpGetUserArg(rxpPnt, RXP_PV_ARG_PV_FMT_NAME)

apply_Programmaopmaak tpPO


On Error Resume Next

  Dim ctl As Control
  For Each ctl In Me.Section(acDetail).Controls
    ctl.FontName = "Arial Narrow"
  Next ctl
End If
