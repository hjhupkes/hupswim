Private Sub handleView(ByVal rptOpenMode As Integer, ByVal rptNCopies As Integer, ByVal rxpFormat As Integer)
On Error GoTo fout


Dim chc As Integer
chc = getChoice

Dim optRnkSmall As Boolean
Dim optRnkShowLevel As Boolean

Dim optRnkRestrictLevel As Boolean
Dim sRnkMinLevel As String
Dim sRnkMaxLevel As String

Dim dRnkLimit As Double

optRnkSmall = Nz(Me.blRnkSmall)
optRnkShowLevel = Nz(Me.blRnkShowLevel)
sRnkMinLevel = Nz(Me.fldRnkMinLevel)
sRnkMaxLevel = Nz(Me.fldRnkMaxLevel)
optRnkRestrictLevel = sRnkMinLevel <> "" Or sRnkMaxLevel <> ""

dRnkLimit = Nz(Me.fldRnkLimiet, 0)

Dim iSOPntMode As Integer
Dim sSOTitle As String
Dim optSoSortAchternaam As Boolean


Dim iMaxRnk As Integer
iMaxRnk = Nz(Me.fldMaxRnk)

iSOPntMode = Nz(Me.optSOPnt)
If (iSOPntMode = PNT_VRT_MODE_NONE) And (chc = OPT_SO) Then
  iMaxRnk = 0
End If
sSOTitle = Nz(Me.fldSOTitle)
optSoSortAchternaam = Nz(Me.blSoSortAchternaam)

Dim optInclude25m As Boolean, optInclude50m As Boolean
Dim optIncludeM As Boolean, optIncludeV As Boolean
Dim optExcludeTst As Boolean, optExcludeDis As Boolean

optInclude25m = Nz(Me.blInclude25m)
optInclude50m = Nz(Me.blInclude50m)
optIncludeM = Nz(Me.blIncludeM)
optIncludeV = Nz(Me.blIncludeV)
optExcludeTst = Not Nz(Me.blIncludeTst)
optExcludeDis = Not Nz(Me.blIncludeDis)

Dim dStartDate As Date, dEndDate As Date
dStartDate = Nz(Me.datum1, #1/1/1900#)
dEndDate = Nz(Me.datum2, #1/1/1900#)



Dim optRestrictLS As Boolean
Dim optOnlyActive As Boolean
Dim optRestrictAfSl As Boolean
Dim optRestrictWed As Boolean
Dim optUseLft As Boolean

optRestrictLS = Nz(Me.blRestrictLS)
optOnlyActive = Nz(Me.blOnlyActive)
If (Not optRestrictLS) Then
  optOnlyActive = False
End If

optRestrictAfSl = Nz(Me.blRestrictAfSl)
optRestrictWed = Nz(Me.blRestrictWed)
optUseLft = Nz(Me.blUseLft)




Dim rptName As String



Select Case chc
  Case OPT_RNK
    If Not rnkBuildRanking(dStartDate, dEndDate, optRestrictAfSl, nAfSl, ordAf, ordSl, _
      optUseLft, nLftCat, lftCat, optRestrictLS, lsCode, optOnlyActive, optRestrictWed, nWeds, ordWeds, _
      optInclude25m, optInclude50m, PNT_VRT_MODE_NONE, optExcludeTst, optExcludeDis, _
      dRnkLimit, optIncludeM, optIncludeV, optRnkRestrictLevel, sRnkMinLevel, sRnkMaxLevel) Then
        Exit Sub
    End If
    
    'Now prepare the report
    If (optRnkSmall) Then
      rptName = REP_RNK_SMALL
    Else
      rptName = REP_RNK
    End If
    
    
  Case OPT_SO
    If Not rnkBuildRanking(dStartDate, dEndDate, optRestrictAfSl, nAfSl, ordAf, ordSl, _
      optUseLft, nLftCat, lftCat, optRestrictLS, lsCode, optOnlyActive, optRestrictWed, nWeds, ordWeds, _
      optInclude25m, optInclude50m, iSOPntMode, optExcludeTst, optExcludeDis, _
      0, optIncludeM, optIncludeV, False, "", "") Then
        Exit Sub
    End If
    
    rptName = REP_SO

End Select

If Not rxpSetRnkOptions(sSOTitle, optRnkShowLevel, iMaxRnk, optSoSortAchternaam) Then
  Exit Sub
End If


Dim vDummy(0 To 0) As Variant


If Not rptHandle(False, rptName, rptOpenMode, rptNCopies, rxpFormat, 0, vDummy) Then
  Exit Sub
End If


Exit Sub
fout:
MsgBox Err & " handleView" & vbCrLf & Error$, 16
Exit Sub

'debug
On Error GoTo 0
Resume Next
End Sub
