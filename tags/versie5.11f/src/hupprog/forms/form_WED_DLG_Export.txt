Private Function suggestFExt(blLXF As Boolean, blWas As Boolean) As String
On Error Resume Next
Dim lxfExt As String
If (blWas) Then
  lxfExt = "xml"
ElseIf (blLXF) Then
  lxfExt = "lxf"
Else
  lxfExt = "lef"
End If
suggestFExt = lxfExt
End Function
Private Function suggestFName(blDlns As Boolean, blTd As Boolean, blLXF As Boolean, blWas As Boolean) As String
On Error GoTo fout
suggestFName = ""


Dim fnPrefix As String

  
  'lxfName = dbdir_op() & "\" & fnPrefix & Format(wed_datum(), "dd-mm-yyyy") & ".lef"

If (blTd Or blWas) Then
  fnPrefix = Format(Nz(wed_datum(), #1/1/1900#), "yyyy\-mm\-dd") & "=" & GoedTekens(Nz(wed_plaats())) & "-" & GoedTekens(Nz(wed_beschrijving())) & "_uitslag"
ElseIf (blDlns) Then
  fnPrefix = Format(Nz(wed_datum(), #1/1/1900#), "yyyy\-mm\-dd") & "=" & GoedTekens(Nz(wed_plaats())) & "-" & GoedTekens(Nz(wed_beschrijving())) & "_deelnemers"
Else
  fnPrefix = Format(Nz(wed_datum(), #1/1/1900#), "yyyy\-mm\-dd") & "=" & GoedTekens(Nz(wed_plaats())) & "-" & GoedTekens(Nz(wed_beschrijving())) & "_programma"
End If


Dim lxfDir As String, lxfExt As String, lxfName As String
lxfDir = dbdir_op()
lxfExt = suggestFExt(blLXF, blWas)
lxfName = directory_goedmaak(dbdir_op()) & "\" & fnPrefix & "." & lxfExt
 
 
suggestFName = lxfName

Exit Function
fout:
MsgBox Err & " " & Error$, 16
Exit Function
End Function

Private Sub btnOK_Click()
On Error GoTo fout
Dim antwoord As Integer, a


Dim expFName As String
expFName = Me.fName

If (expFName = "") Then
  MsgBox "Kies een bestandsnaam.", 16
  Exit Sub
End If


Dim isLx As Boolean
isLx = (Nz(Me.mode) = 20)

Dim isWas As Boolean
isWas = (Nz(Me.mode) = 6)


If (isLx) Then
  
  
  If Not checkDepotnummers(waarde()) Then
     Exit Sub
  End If
  
  Dim blDlns As Boolean
  Dim blTd As Boolean
  Dim blLXF As Boolean
  blDlns = Nz(Me.blLenDeelnemers)
  blTd = Nz(Me.blLenTijden)
  blLXF = Nz(Me.blLXF)
  
  
  Dim lxfDir As String, lxfExt As String, lxfName As String
  
  
  lxfDir = extractFilePath(expFName)
  lxfName = extractFileName(expFName)
  
  Dim wednr As Long
  If Me.blMeet And wed_IsMeet() Then
    wednr = wed_MeetMainNr
  Else
    wednr = waarde()
  End If
  
  Dim lxVersion As Long
  lxVersion = LX_VERSION_2
  If (Nz(Me.blVersion3)) Then
    lxVersion = LX_VERSION_3
  End If
  
  
  
  If (lenHandleCompleteExportSequence(lxfDir, lxfName, Not Nz(Me.blMeet), wednr, Not (Me.blLenExt), lxVersion, blDlns, blTd)) Then
    DoCmd.Close
  End If
 

  Exit Sub
End If

If (isWas) Then
  
  If Not checkDepotnummers(waarde()) Then
    Exit Sub
  End If
  
  If (lenHandleCompleteWasExportSequence(expFName, WAS_FILE_TYPE_UIT, waarde(), True)) Then
    DoCmd.Close
  End If
 

  Exit Sub
End If



Exit Sub

fout:
MsgBox Err & " btnOK_click" & vbCrLf & Error$, 16
Exit Sub

End Sub

Private Sub btnWijzig_Click()
On Error GoTo fout


Dim expFName As String

expFName = Nz(Me.fName)


Dim blDlns As Boolean
  Dim blTd As Boolean
  Dim blLXF As Boolean
  Dim blWas As Boolean
  
  blDlns = Nz(Me.blLenDeelnemers)
  blTd = Nz(Me.blLenTijden)
  blLXF = Nz(Me.blLXF)
  
  blWas = Nz(Me.mode) <> 20

If Not DirectoryExists(extractFilePath(expFName)) Then
  If Not DirectoryExists(dbdir_op()) Then
    expFName = "c:\" & extractFileName(suggestFName(blDlns, blTd, blLXF, blWas))
  Else
    expFName = directory_goedmaak(dbdir_op()) & extractFileName(suggestFName(blDlns, blTd, blLXF, blWas))
  End If
End If
  
Dim fExt As String
fExt = suggestFExt(blLXF, blWas)
  
If (fExt = "xml") Then
  If Not fileNamePromptUser(FILE_DLG_SAVE, "Was2005 uitslag Exporteren ", "Was2005 bestand(*.xml)|*.xml", expFName, True, True) Then
    Exit Sub
  End If
ElseIf (fExt = "lef") Then
  If Not fileNamePromptUser(FILE_DLG_SAVE, "Lenex bestand aanmaken ", "Lenex (*.lef)|*.lef", expFName, True, True) Then
    Exit Sub
  End If
ElseIf (fExt = "lxf") Then
  If Not fileNamePromptUser(FILE_DLG_SAVE, "Lenex bestand aanmaken ", "Lenex (*.lxf)|*.lxf", expFName, True, True) Then
    Exit Sub
  End If
End If

If (expFName <> "") Then
  If (Right(expFName, 4) <> "." & fExt) Then
    expFName = expFName & "." & fExt
  End If
  Me.fName = expFName
End If


Exit Sub
fout:
MsgBox Err & " " & Error$, 16
End Sub
