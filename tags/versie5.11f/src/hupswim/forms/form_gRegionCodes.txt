Private Sub selectAll(sel As Boolean)
On Error Resume Next
Dim rs As Recordset
Set rs = Me.RecordsetClone


DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20

rs.MoveFirst
Do Until rs.EOF
  rs.Edit
  rs("RG_Import") = sel
  rs.Update
  rs.MoveNext
Loop

Me.Requery

End Sub
