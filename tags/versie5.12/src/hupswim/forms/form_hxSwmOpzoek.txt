Private sVer As String
Private gOnlyActive As Boolean

Private Function perform() As Boolean
On Error GoTo fout
perform = False

Dim tpStart As String
tpStart = Nz(Me.lijst.value)

If tpStart = "" Then
  Exit Function
End If

Dim rs As Recordset
Set rs = CurrentDb().OpenRecordset("gSwmOpzoek")

rs.MoveFirst

If (rs.EOF) Then
  Err.Raise 1, , "internal error."
End If

rs.Edit
  rs("SO_Startnummer") = hzn(tpStart)
  rs("SO_OK") = True
rs.Update

appCleanRS rs

perform = True
Exit Function
fout:
If (Err = 3021) Then Resume Next ' no current record
MsgBox Err & " Perform" & vbCrLf & Error$, 16
Exit Function
End Function


Private Sub initView()
On Error Resume Next
sVer = Nz(DLookup("SO_ARG_Vereniging", "gSwmOpzoek"))
gOnlyActive = Nz(DLookup("SO_ARG_ActiefLid", "gSwmOpzoek"))
End Sub


Private Sub updateView()
On Error Resume Next

Me.blOnlyActive.Visible = Not gOnlyActive

Dim sqlA As String, sqlS As String, sqlL As String

sqlA = "select startnummer, [achternaam] & "", "" & [voornaam] & "" "" &  [voegsel] as NM from [dtLeden]  "
sqlL = "select startnummer, fmtName(nz([voornaam]),nz([voegsel]),nz([achternaam])) as NM, geslacht, [actief lid] from [dtLeden] "

If (gOnlyActive Or Nz(Me.blOnlyActive)) Then
  sqlA = sqlA & " where [actief lid] = true and startnummer <> ""00-000"" "
  sqlL = sqlL & " where [actief lid] = true and startnummer <> ""00-000"""
Else
  sqlA = sqlA & " where  startnummer <> ""00-000"" "
  sqlL = sqlL & " where startnummer <> ""00-000"" "
End If
sqlS = sqlA



sqlS = sqlS & " order by stnrExpand(nz([startnummer]));"
sqlA = sqlA & " order by achternaam;"

If (Nz(Me.blSortAchternaam)) Then
  sqlL = sqlL & " order by achternaam;"
Else
  sqlL = sqlL & " order by stnrExpand(nz([startnummer]));"
End If


Me.fldStartnummer.RowSource = sqlS
Me.fldAchternaam.RowSource = sqlA
Me.lijst.RowSource = sqlL

Me.fldStartnummer.Requery
Me.fldAchternaam.Requery
Me.lijst.Requery



End Sub
