Private Sub blUseZip_AfterUpdate()
On Error Resume Next

Dim fName As String

fName = Me.bckFName.Caption


Dim fExt As String
If Nz(Me.blUseZip) Then
  fExt = "zip"
Else
  fExt = "wed"
End If

If (fName <> "") Then
  If (Right(fName, 4) <> ".zip") And (Right(fName, 4) <> ".wed") Then
    fName = fName & "." & fExt
  ElseIf (Right(fName, 4) = ".wed") And Nz(Me.blUseZip) Then
    fName = Left(fName, Len(fName) - 4) & ".zip"
  ElseIf (Right(fName, 4) = ".zip") And Not Nz(Me.blUseZip) Then
    fName = Left(fName, Len(fName) - 4) & ".wed"
  End If
  
  Me.bckFName.Caption = fName
End If



End Sub

Private Sub btnWijzig_Click()
On Error Resume Next

Dim fName As String

fName = Me.bckFName.Caption


Dim fExt As String
If Nz(Me.blUseZip) Then
  fExt = "zip"
Else
  fExt = "wed"
End If

If Not DirectoryExists(extractFilePath(fName)) Then
  fName = directory_goedmaak(dbdir_op()) & "\Wed" & Format(Nz(wed_datum(), #1/1/1900#), "yyyy\-mm\-dd") & "." & fExt
End If
  
Dim fExtFilter As String

If (fExt = "zip") Then
  fExtFilter = "Wedstrijd backup (wed*.zip)|*.zip"
Else
  fExtFilter = "Wedstrijd backup (wed*.wed)|*.wed"
End If
  
If Not fileNamePromptUser(FILE_DLG_SAVE, "Wedstrijd backup aanmaken ", fExtFilter, fName, True, True) Then
  Exit Sub
End If

If (fName <> "") Then
  If (Right(fName, 4) <> ".zip") And (Right(fName, 4) <> ".wed") Then
    fName = fName & "." & fExt
  ElseIf (Right(fName, 4) = ".wed") And Nz(Me.blUseZip) Then
    fName = Left(fName, Len(fName) - 4) & ".zip"
  ElseIf (Right(fName, 4) = ".zip") And Not Nz(Me.blUseZip) Then
    fName = Left(fName, Len(fName) - 4) & ".wed"
  End If
  
  Me.bckFName.Caption = fName
End If

End Sub

Private Sub Form_Load()
On Error Resume Next

Dim fName As String

Dim fExt As String
If Nz(Me.blUseZip) Then
  fExt = "zip"
Else
  fExt = "wed"
End If

fName = ""
If DirectoryExists(dbdir_backup()) Then
  fName = directory_goedmaak(dbdir_op()) & "\Wed" & Format(Nz(wed_datum(), #1/1/1900#), "yyyy\-mm\-dd") & "." & fExt
Else
  fName = "<Uw backup directory is ongeldig. Wijzig deze in het Instellingen scherm>"
End If

Me.bckFName.Caption = fName


End Sub
