Private Function intAppend(x1, x2, x3) As String
If (Nz(x1, "") = "") Then
  intAppend = Nz(x3, "")
Else
  If (Nz(x3, "") = "") Then
    intAppend = x1

    intAppend = Nz(x1, "") & Nz(x2, "") & Nz(x3, "")
  End If
End If

Private Function Plaats(x) As String
On Error GoTo hieruit
If IsNull(x) = True Then
    Plaats = Null
    Exit Function
End If
If x = "1" Then
    Plaats = "I"
    Exit Function
End If
If x = "2" Then
    Plaats = "II"
    Exit Function
End If
If x = "3" Then
    Plaats = "III"
    Exit Function
Else
    Plaats = x
End If

hieruit:
Plaats = ""
Exit Function
End Function

Private Sub updateWedData()
On Error GoTo fout

Dim tpWedNr As Long
tpWedNr = Nz(Me.[PR_Wedstrijdnummer])
    
If (tpWedNr <> lRecCurWedNr) Then

  Dim dummy(0 To 0) As String

  If (Not catInfoGetListForWed(tpWedNr, nMCatProgs, mCatProgNames, dummy, mCatProgOffs, mCatProgCatCnt, catInfo, False)) Then
    nMCatProgs = 0
    Exit Sub
  End If
  
  
  If (Not prGetExtRecForWed(tpWedNr, nExtRecProgs, extRecProgNames, extRecProgOffs, extRecProgCnt, extRecInfo)) Then
    nExtRecProgs = 0
    Exit Sub
  End If
  
  If (Not prGetWinForWed(tpWedNr, nWinProgs, winProgNames, winProgOffs, winProgCnt, winInfo)) Then
    nWinProgs = 0
    Exit Sub
  End If
  
  lRecCurWedNr = tpWedNr
  
End If
    
Exit Sub
fout:
MsgBox Err & " updateWedData" & vbCrLf & Error$, 16
Exit Sub
End Sub

If (FormatCount = 1) Then

      Exit Sub
    End If
    
    Dim isEmpty As Boolean
    isEmpty = True
    
    Dim isLngOpmEmpty As Boolean
    Dim isCatListEmpty As Boolean
    
    isLngOpmEmpty = True
    isCatListEmpty = True

      Me.fldCatListOpmHelper.Height = Me.scProgHeaderOpmCatList.Height - szEmptyLineCM * 567 - Me.fldCatListOpmHelper.Top

    If (blIsUitslagLijst) Then
      If blSkipCurDeeln Then
        Cancel = True
        Exit Sub
      End If
    End If
    
    'Collect information
    Dim topHeight As Long
    topHeight = 0
    'Dim estGrpFooter As Section
    
    'pocGetEstafOpmaakInfo topHeight, estGrpFooter
    
    
    Dim blIsNoData As Boolean
    blIsNoData = True
    
    If Not fPO.blEstafsIntern Or (Not Nz(Me.PR_Afstand) Like "*x*") Then
      blIsNoData = True
      If (topHeight = 0) Then
        'we are allowed to cancel

      End If

      Dim estWed As Long
      Dim estProg As String
      Dim estNr As Integer
      Dim estVer As String

      estProg = Nz(Me.PR_Programmanummer)

    If (Not Cancel) Then
        Dim maxUsedHeight As Long

        For i = 1 To MAX_ESTAF_LIST_CNT

          blIsEmpty = True

Private Sub scDetail_Format(Cancel As Integer, FormatCount As Integer)
On Error GoTo fout


If IsNull([DE_Vereniging]) Then
  
  Dim i As Integer
  For i = 0 To nDtlCtls - 1
    If ctlDtl(i).Visible Then
      ctlDtl(i).Visible = False
    End If
  Next i
  Me.fldNoDlns.Visible = True
  

Else

  
  Dim curDisCode As String
  If (blIsUitslagLijst Or blIsUitslag) Then
    curDisCode = Nz([ty_dis])
  End If
  
  
  
  If (blIsUitslagLijst) Then
  
    Dim blIsPntProg As Boolean
    blIsPntProg = Nz(Me.Controls("PR_Type")) = PROG_TYPE_PUNT
  
    blSkipCurDeeln = False
  
    Dim deelnIsValid As Boolean
    deelnIsValid = True
    
    If (Not blIsPntProg) Then
      If Not swtIsValid(Nz([TY_Eindtijd])) Then
        deelnIsValid = False
      End If
    Else
      'Automatically valid
      If (Abs(Nz(Me.Controls("TY_EindPunten"))) < 0.5) Then
        deelnIsValid = False
      End If
    End If
    
    If (curDisCode <> "") Then
      If (Not (Nz([PR_DisTeltMee], False))) Then
        deelnIsValid = False
      End If
    End If
    
    If (Nz([DE_Buitenmededinging]) = "J") Then
      deelnIsValid = False
    End If
      
    If intOnlyFirstRes <> -1 Then
      If (Not deelnIsValid) Or Nz(Me.Controls("TY_EchtePlaats").value) > intOnlyFirstRes Then
        blSkipCurDeeln = True
        Cancel = True
        Exit Sub
      End If
    End If
    'Me.Controls("fldTeller").Visible = deelnIsValid
  
  
  
  
  ElseIf (blIsStartLijst) Then
  
  
    If (Not blIsOW) Then
      deelnIsValid = True
      If (Nz([DE_Buitenmededinging]) = "J") Then
        deelnIsValid = False
      End If
    
      
      Me.Controls("cnt").Visible = deelnIsValid
    End If
  
  
  End If
  
  
  
  If (blIsUitslag Or blIsUitslagLijst) Then
    If curDisCode = "" Then
      Me.Controls("pla").Visible = True
      Me.Controls("dis").Visible = False
    Else
      Me.Controls("pla").Visible = False
      Me.Controls("dis").Visible = True
      
      If Len(curDisCode) > 2 Then
          'Only move if isUitslag.
          If (blIsUitslag) Then
            If Len(curDisCode) > 4 Then
                Me.Controls("dis").Left = o_left_dis - 0.3 * 567
                Me.Controls("dis").Width = o_width_dis + 0.3 * 567
            Else
               Me.Controls("dis").Width = o_width_dis
               Me.Controls("dis").Left = o_left_dis
            End If
          End If
          Me.Controls("dis").FontName = "Arial narrow"
          Me.Controls("dis").FontSize = 8
      Else
          Me.Controls("dis").Width = o_width_dis
          Me.Controls("dis").Left = o_left_dis
          
          Me.Controls("dis").FontName = o_font_dis.name
          Me.Controls("dis").FontSize = o_font_dis.Size
      End If
    End If
  End If 'if blIsUitslag or blIsUitslagLijst
  
End If 'if isnull([de_vereniging])

'If (blIsStartLijst) Then
'  pobFormatEstafSection Cancel, FormatCount
'End If

Exit Sub
fout:
MsgBox Err & " scDetail_OnFormat" & vbCrLf & Error$, 16
Exit Sub
End Sub
