Private Sub btnDeselecteren_Click()
On Error Resume Next
Dim geg As Recordset
Set geg = Me.Form.RecordsetClone
geg.MoveFirst
Do Until geg.EOF
    geg.Edit
    geg("LSS_IsSelected") = False
    geg.Update
    geg.MoveNext
Loop
Me.Form.Requery
End Sub

Private Sub btnOK_Click()
On Error Resume Next
'Have to make sure that isOK is set to true

Dim rs As Recordset
Set rs = CurrentDb().OpenRecordset("hpoLSSimpleOpzoek")
rs.MoveFirst
  
  rs.Edit
    rs("LSS_OK") = True
  rs.Update

appCleanRS rs
    

DoCmd.Close


End Sub

End Sub
