Private Sub btnSelNone_Click()
On Error Resume Next
Dim geg As Recordset
Set geg = Me.Form.RecordsetClone
geg.MoveFirst
Do Until geg.EOF
    geg.Edit
    geg("IMP_Import") = False
    geg.Update
    geg.MoveNext
Loop

End Sub
