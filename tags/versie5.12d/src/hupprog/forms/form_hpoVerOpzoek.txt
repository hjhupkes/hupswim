Private Function perform() As Boolean
On Error GoTo fout
perform = False

Dim vName As String
vName = Nz(Me.lijst.Column(0))

If (vName = "") Then
  fldVerName.SetFocus
  Exit Function
End If


Dim rs As Recordset

Set rs = CurrentDb().OpenRecordset("hpoVerOpzoek")

rs.MoveFirst

If (rs.EOF) Then
  Err.Raise 1, , "Internal error. Tabel hpoVerOpzoek is leeg."
End If

rs.Edit
  rs("PO_OK") = True
  rs("PO_Vereniging") = vName
rs.Update

appCleanRS rs

perform = True
Exit Function
fout:
If (Err = 3021) Then Resume Next
MsgBox Err & " perform" & vbCrLf & Error$, 16
Exit Function
End Function
