Private Sub btnDeselAll_Click()
On Error Resume Next
DoCmd.DoMenuItem A_FORMBAR, A_FILE, A_SAVERECORD, , A_MENU_VER20
Dim geg As Recordset
Set geg = Me.Form.RecordsetClone
geg.MoveFirst
Do Until geg.EOF
    geg.Edit
    geg("importeren") = False
    geg.Update
    geg.MoveNext
Loop
Me.Form.Requery
End Sub
