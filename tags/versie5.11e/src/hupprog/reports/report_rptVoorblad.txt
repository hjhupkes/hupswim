Private sDlnVers As String


Private Function initReport() As Boolean
On Error GoTo fout
initReport = False

Dim blShowDlnVers As Boolean
blShowDlnVers = Nz(DLookup("VBLD_ShowVers", "~voorblad_stats"), True)

If (blShowDlnVers) Then
  sDlnVers = Nz(DLookup("VBLD_Vers", "~voorblad_stats"))
Else
  sDlnVers = "Zie bijlage"
End If


initReport = True
Exit Function
fout:
MsgBox Err & " initReport" & vbCrLf & Error$, 16
Exit Function
End Function


Private Sub scDetail_Format(Cancel As Integer, FormatCount As Integer)
On Error GoTo fout


Me![line1].Visible = True
Me![line2].Visible = True
Me![ed].Visible = True
Me![klok].Visible = True

Dim blKlokTimeEntered As Boolean
blKlokTimeEntered = Nz(Me![JUX_KlokTimeEntered])
  

If IsNull(Me![ed]) Or Me![ed] = "" Then
    Me![line1].Visible = Not blKlokTimeEntered
    Me![ed].Visible = False
Else
    Me![line1].Visible = False
    Me![ed].Visible = True
End If
If IsNull(Me![klok]) Or Nz(Me![klok]) = "" Then
    Me![line2].Visible = Not blKlokTimeEntered
    Me![klok].Visible = False
Else
    Me![line2].Visible = False
    Me![klok].Visible = True
End If

Exit Sub

fout:
MsgBox Err & " scDetail_format" & vbCrLf & Error$, 16
Exit Sub

End Sub






Private Sub Report_Open(Cancel As Integer)
On Error Resume Next
If Not initReport() Then

End If
End Sub
