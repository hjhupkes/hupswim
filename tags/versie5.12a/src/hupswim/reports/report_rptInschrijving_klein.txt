On Error GoTo fout

Dim rs As Recordset

rs.MoveFirst

cntV = Nz(rs("OP_CntDames"))
cntM = Nz(rs("OP_CntHeren"))
cntPloeg = Nz(rs("OP_CntStartsPloegen"))

dPerStartDate = Nz(rs("OP_PeriodStartDate"))
dPerEndDate = Nz(rs("OP_PeriodEndDate"))

appCleanRS rs


Set rs = CurrentDb().OpenRecordset("select * from [~Opstelling] order by op_wedstrijdnummer")

rs.MoveFirst



strBeschrijving = Nz(rs("OP_WedBeschrijving"))
strZwembad = Nz(rs("OP_WedZwembad"))

Dim tpWedDate As Date
tpWedDate = Nz(rs("OP_WedDatum"))

strDatum = ""
If (lenDateIsValid(tpWedDate)) Then
  strDatum = Format(tpWedDate, "dd\-mm\-yyyy")
End If

strAanvang = ""
If Not IsNull(rs("OP_WedAanvang")) Then
  strAanvang = Format(rs("OP_WedAanvang"), "hh:nn")
End If

strPloegleider = Nz(rs("OP_WedPloegleider"))

appCleanRS rs





mCatCurWedNr = -1



Exit Sub
fout:
If (Err = 3021) Then Resume Next
MsgBox Err & " initReport" & vbCrLf & Error$, 16
Exit Sub

  prognr = Nz([OP_Programmanummer])
  Dim progIdx As Integer
  progIdx = catInfoGetProgIdx(prognr, nMCatProgs, mCatProgNames)
  If (progIdx >= 0) Then
    Me.extBesProg.value = fmtCatLijst(catInfo, mCatProgOffs(progIdx), mCatProgCatCnt(progIdx))
    captSet = True
    
    Dim i As Integer
    Dim lastRecIdx As Integer
    lastRecIdx = 0
    For i = 0 To mCatProgCatCnt(progIdx) - 1
      If (Len(Trim(catInfo(mCatProgOffs(progIdx) + i).rec.recType)) > 0) Then
        With catInfo(mCatProgOffs(progIdx) + i).rec
          lastRecIdx = lastRecIdx + 1
          Me.Report.Controls("recType" & lastRecIdx).value = Trim(.recType)
          Me.Report.Controls("recNaam" & lastRecIdx).value = Trim(.naam)
          Me.Report.Controls("recTijd" & lastRecIdx).value = fmtSwimTime(.Tijd)
          Me.Report.Controls("recDatum" & lastRecIdx).value = fmtDateShort(.Datum)
          Me.Report.Controls("recPlaats" & lastRecIdx).value = Trim(.Plaats)
          Me.Report.Controls("recCat" & lastRecIdx).value = Trim(catInfo(mCatProgOffs(progIdx) + i).cat)
        End With
      End If
    Next i
    For i = lastRecIdx + 1 To MAX_N_VAR_CATS_PER_PROG
          Me.Report.Controls("recType" & i).value = ""
          Me.Report.Controls("recNaam" & i).value = ""
          Me.Report.Controls("recTijd" & i).value = ""
          Me.Report.Controls("recDatum" & i).value = ""
          Me.Report.Controls("recPlaats" & i).value = ""
          Me.Report.Controls("recCat" & i).value = ""
    Next i
  End If
End If
If Not captSet Then
  Cancel = True
End If

End Sub
