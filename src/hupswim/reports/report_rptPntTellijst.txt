On Error GoTo fout

cumTvDis = 0

"[__punten].PT_Is_Estimate , [__punten].PT_Is_InitVals, [__punten].PT_Vereniging, [__punten].PT_Depot, " & _
"[__punten].PT_Programmanummer , [__punten].PT_Afstand, [__punten].PT_slag, [__punten].PT_categorie, " & _
"[__punten].PT_Beschrijving_categorie , [__punten].PT_Beschrijving_afwijkend, [__punten].PT_Type, " & _
"[__punten].PT_N_Meetellende_zwemmers , [__punten].PT_Vervangende_tijd, [__punten].PT_Is_Estaf, " & _
"[__punten].PT_Startnummer_1 , [__punten].PT_Naam_1, [__punten].PT_Tijd_1, [__punten].PT_Actual_Tijd_1, " & _

"[__punten].PT_Estafettenummer_1 , [__punten].PT_INT_EstGUID_1, [__punten].PT_Startnummer_2, " & _
"[__punten].PT_Naam_2 , [__punten].PT_Tijd_2, [__punten].PT_Actual_Tijd_2, [__punten].PT_Toevoegen_dis_2, " & _
"[__punten].PT_Is_Dis_2, [__punten].PT_IS_VVT_2, [__punten].PT_Punten_2, [__punten].PT_Estafettenummer_2, " & _

Private Sub initFormat()
On Error GoTo fout

If db_narrow() Then
  Dim ctl As Control
  For Each ctl In Me.Section(acDetail).Controls
    If (ctl.ControlType = acTextBox) Then
      ctl.FontName = "Arial Narrow"
    End If
  Next ctl
End If


Exit Sub
fout:
MsgBox Err & " initFormat" & vbCrLf & Error$, 16
Exit Sub
End Sub

Private Sub scDetail_Format(Cancel As Integer, FormatCount As Integer)
On Error Resume Next
If (Nz(Me.[PT_Toevoegen_dis_Cumulatief]) > cumTvDis) Then
  cumTvDis = Nz(Me.[PT_Toevoegen_dis_Cumulatief])
End If
If (Nz(Me.[PT_Cumulatief]) > cumPnt) Then
  cumPnt = Nz(Me.[PT_Cumulatief])
End If
If (Nz(Me.[PT_Cumulatief_NoInitVal]) > cumPntNoInitVal) Then
  cumPntNoInitVal = Nz(Me.[PT_Cumulatief_NoInitVal])
End If


End Sub




Private Sub scRepFooter_Format(Cancel As Integer, FormatCount As Integer)
On Error Resume Next
Me.pntSubtot = cumPntNoInitVal - cumTvDis
Me.pntTot = cumPntNoInitVal
Me.tvDisTot = cumTvDis

If (cumPnt - cumPntNoInitVal > 0.0001) Then
  Me.lbGrandTotal.Visible = True
  Me.lbInitVal.Visible = True
  Me.pntGrandTotal.Visible = True
  Me.pntInitVal.Visible = True
  
  Me.pntInitVal = cumPnt - cumPntNoInitVal
  Me.pntGrandTotal = cumPnt
End If
End Sub

Private Sub Report_Open(Cancel As Integer)
On Error Resume Next
initReport
initFormat
End Sub

End Sub
