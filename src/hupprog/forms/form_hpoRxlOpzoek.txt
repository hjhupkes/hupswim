Private Function perform() As Boolean
On Error GoTo fout
perform = False

Dim lObjID As Long
Dim sObjDes As String
Dim blIsNew As Boolean


blIsNew = False

Dim blAllowNew As Boolean
blAllowNew = Nz(DLookup("RXLO_ARG_AllowNewName", "hpoRxlOpzoek"))

If (Nz(Me.lijst, "") = "") Then
  If (blAllowNew) And Nz(Me.fldObj.Column(1)) = "" Then
    blIsNew = True
    sObjDes = Nz(Me.fldObj)
    lObjID = -1
  End If
Else
  lObjID = Nz(Me.lijst.Column(0))
  sObjDes = Nz(Me.lijst.Column(1))
End If


If (sObjDes = "") Then
  fldObj.SetFocus
  Exit Function
End If


Dim rs As Recordset

Set rs = CurrentDb().OpenRecordset("hpoRxlOpzoek")

rs.MoveFirst

If (rs.EOF) Then
  Err.Raise 1, , "Internal error. Tabel hpoRxlOpzoek is leeg."
End If

rs.Edit
  rs("RXLO_OK") = True
  rs("RXLO_ObjID") = lObjID
  rs("RXLO_ObjDes") = hzn(sObjDes)
  rs("RXLO_IsNew") = blIsNew
rs.Update

appCleanRS rs

perform = True
Exit Function
fout:
If (Err = 3021) Then Resume Next
MsgBox Err & " perform" & vbCrLf & Error$, 16
Exit Function

'debug
On Error GoTo 0
Resume

End Function


Private Sub initView()
On Error GoTo fout

Dim blAllowNew As Boolean
Dim blUseCurObjID As Boolean
Dim lCurObjID As Long
Dim blUseCurObjDes As Boolean
Dim sCurObjDes As String

Dim sDlgTitle As String

Dim lObjTypeID As Long

Dim iMinVisLevel As Integer


blAllowNew = Nz(DLookup("RXLO_ARG_AllowNewName", "hpoRxlOpzoek"))
blUseCurObjID = Nz(DLookup("RXLO_ARG_UseCurObjID", "hpoRxlOpzoek"))
lCurObjID = Nz(DLookup("RXLO_ARG_CurObjID", "hpoRxlOpzoek"))
blUseCurObjDes = Nz(DLookup("RXLO_ARG_UseCurObjDes", "hpoRxlOpzoek"))
sCurObjDes = Nz(DLookup("RXLO_ARG_CurObjDes", "hpoRxlOpzoek"))
sDlgTitle = Nz(DLookup("RXLO_ARG_DlgTitle", "hpoRxlOpzoek"))
lObjTypeID = Nz(DLookup("RXLO_ARG_ObjTypeID", "hpoRxlOpzoek"))
iMinVisLevel = Nz(DLookup("RXLO_ARG_MinVisibilityLevel", "hpoRxlOpzoek"))

If (blAllowNew) Then
  Me.fldObj.LimitToList = False
  Me.lbNewAllowed.Visible = True
Else
  Me.fldObj.LimitToList = True
  Me.lbNewAllowed.Visible = False
End If

If (blUseCurObjID) Then
  Dim tpObjDes As String
  tpObjDes = Nz(DLookup("ro_description", "hpoRptExpListObjects", "ro_id=" & lCurObjID), -1)
  Me.fldObj = tpObjDes
  Me.lijst = lCurObjID
ElseIf (blUseCurObjDes) Then
  Dim tpObjID As Long
  tpObjID = Nz(DLookup("ro_id", "hpoRptExpListObjects", "ro_description=""" & sCurObjDes & """"), -1)
  Me.fldObj = sCurObjDes
  Me.lijst = tpObjID
End If

If (sDlgTitle <> "") Then
  Me.lbTitle.Caption = sDlgTitle
  Me.Caption = sDlgTitle
Else
  Me.lbTitle.Caption = "Kies naam"
  Me.Caption = "Kies naam"
End If



Dim listSql As String, cmbSql As String

listSql = "select distinctrow ro_id, ro_description from hpoRptExpListObjects where ro_type_id = " & lObjTypeID & _
  " and ro_visibilityLevel >= " & iMinVisLevel & _
  " order by ro_order, ro_id"
      
      
'Need to swith column order for combo box to allow new values to be typed into the combo
cmbSql = "select distinctrow  ro_description, ro_id from hpoRptExpListObjects where ro_type_id = " & lObjTypeID & _
  " and ro_visibilityLevel >= " & iMinVisLevel & _
  " order by ro_order, ro_id"
      
      
Me.fldObj.RowSource = cmbSql
Me.lijst.RowSource = listSql

Exit Sub
fout:
MsgBox Err & " initView" & vbCrLf & Error$, 16
Exit Sub

End Sub
